// ① 썸네일 이미지를 Multipart/form-data로 서버 업로드
void AHttpActor_KMK::ReqMultipartCapturedURL( FStageInfo& Stage, const FString& ImagePath )
{
    TSharedRef<IHttpRequest> Req = FHttpModule::Get().CreateRequest();
    Req->SetURL( TEXT("http://back.reward-factory.shop:8123/api/v1/files/upload") );
    Req->SetVerb( TEXT("POST") );

    TArray<uint8> FileData;
    FFileHelper::LoadFileToArray( FileData, *ImagePath );

    FString Boundary = "---------------------------boundary_string";
    FString FileHeader = "Content-Disposition: form-data; name=\"file\"; filename=\""
                         + FPaths::GetCleanFilename(ImagePath)
                         + TEXT("\"\r\nContent-Type: image/png\r\n\r\n");

    TArray<uint8> PostData;
    PostData.Append( (uint8*)TCHAR_TO_ANSI(*(FString("--") + Boundary + "\r\n")) , ... );
    PostData.Append( (uint8*)TCHAR_TO_ANSI(*FileHeader), FileHeader.Len() );
    PostData.Append( FileData );

    Req->SetHeader( TEXT("Content-Type"),
        FString::Printf(TEXT("multipart/form-data; boundary=%s"), *Boundary) );
    Req->SetContent( PostData );
    Req->ProcessRequest();
}

void AHttpActor_KMK::ReqStageInfo( const FStageInfo& Stage )
{
    FString JsonString;
    FJsonObjectConverter::UStructToJsonObjectString( Stage, JsonString );

    TSharedRef<IHttpRequest> Req = FHttpModule::Get().CreateRequest();
    Req->SetURL( TEXT("http://back.reward-factory.shop:8123/api/v1/stages") );
    Req->SetVerb( TEXT("POST") );
    Req->SetHeader( TEXT("Content-Type"), TEXT("application/json") );
    Req->SetContentAsString( JsonString );
    Req->ProcessRequest();
}

1. 테트리스형 2D 그리드 인벤토리
  • InventoryModel이 InventoryItem[,] _grid 2D 배열로 상태를 관리, 이벤트(OnItemPlaced / OnItemRemoved)로 View에 통보하는 이벤트 기반 MVC 패턴 설계
  • 아이템마다 InventorySize(Vector2Int)와 IsRotated 플래그를 두어 90도 회전 시 EffectiveSize의 x/y를 교환 — 배치 충돌 판정에 그대로 반영
  • TryPlace / CanPlace 메서드가 차지하는 모든 셀을 순회해 공간 충돌을 원자적으로 검사
  • OutsideInventoryAreaView로 그리드 밖 좌·우 인벤토리 영역을 독립 관리 — 그리드 경계에서 방향키 입력 시 Outside 영역으로 포커스 자동 전환
  • InventoryDragPreviewView가 드래그 중 아이템 실루엣과 배치 가능 여부 하이라이트를 실시간 렌더링
// 아이템 배치 — 차지하는 모든 셀 충돌 검사 후 원자적 배치
public bool TryPlace(InventoryItem item, Vector2Int pos)
{
    if (!CanPlace(item, pos)) return false;

    item.Place(pos);
    foreach (var cell in item.GetOccupiedCells())
        _grid[cell.x, cell.y] = item;

    OnItemPlaced?.Invoke(item);
    return true;
}

// 회전 시 EffectiveSize x/y 교환으로 충돌 판정 자동 반영
public Vector2Int EffectiveSize
    => IsRotated ? new Vector2Int(Size.y, Size.x) : Size;
2. 마우스 · 게임패드 하이브리드 입력 (InventoryState)
  • Mouse.current.delta.sqrMagnitude > 0.1f 감지로 마우스 모드 전환, 게임패드 버튼 입력 시 패드 모드로 즉시 전환 — 하나의 인벤토리가 두 입력장치를 실시간 병행 지원
  • 스틱 아날로그 입력을 디지털 4방향으로 변환하고 초기 딜레이(0.4s) + 반복 간격(0.12s) 타이머로 키 반복 네비게이션 구현
  • 그리드 좌·우 경계에서 방향 입력 시 Outside 영역으로, 반대 방향 입력 시 그리드 가장자리 셀로 포커스 자동 이동하는 스마트 네비게이션
  • 패드 A 버튼 — 토글 방식으로 집기·놓기, LB/RB — 드래그 중 아이템 90도 회전, B 버튼 — 드래그 취소 및 원위치 복구
// 마우스 델타 · 게임패드 버튼 감지로 실시간 입력 모드 전환
private void UpdateInputMode()
{
    if (IsMouseMoving())
        SwitchMode(InputMode.Mouse);
    else if (IsGamepadActive())
        SwitchMode(InputMode.Gamepad);
}

// 스틱 반복 입력 — 초기 딜레이 후 고속 반복 네비게이션
private void UpdateStickRepeat()
{
    if (_stickDirection == Vector2Int.zero) return;
    _stickRepeatTimer -= Time.deltaTime;
    if (_stickRepeatTimer > 0f) return;

    NavigateInDirection(_stickDirection);
    _stickRepeatTimer = STICK_REPEAT_INTERVAL; // 0.12s
}
3. 장소 기반 퀘스트 시스템
인스펙터 데이터 구조 — 코드 수정 없이 데이터만 바꿔 새 퀘스트 제작 가능
Quest (ScriptableObject)
├─ TaskGroup[0]
│   ├─ QuestTask  category: "DELIVERY_ACCEPT"  target: A장소
│   └─ QuestAction  → Yarn 대화 노드 변경
└─ TaskGroup[1]
    └─ QuestTask  category: "DELIVERY_COMPLETE"  target: B장소
런타임 흐름
1
씬 시작 — 퀘스트 등록
씬에 배치된 QuestGiver 컴포넌트가 QuestManager.Register(quest) 호출
→ Quest를 Clone해 _activeQuests에 추가, state: Inactive → Running, 첫 TaskGroup 시작
2
플레이어 행동 — 배달 수락
플레이어가 A장소에서 배달을 수락 → DeliveryManager.AcceptDelivery() 내부에서
QuestManager.ReceiveReport("DELIVERY_ACCEPT", A장소, 1) 호출
3
QuestManager — 전체 브로드캐스트
_activeQuests를 순회하며 모든 활성 퀘스트에 quest.ReceiveReport() 전달
→ 매칭 여부와 상관없이 일단 전부 찔러봄. 맞는 것만 반응
4
QuestTask — 카테고리 · 타겟 매칭
각 QuestTask가 IsTarget(category, target)으로 자신에게 해당되는지 확인
인스펙터에서 category = "DELIVERY_ACCEPT", target = A장소로 설정된 Task만 반응
5
TaskGroup 완료 → QuestAction 실행
CurrentSuccess++ → needSuccess 달성 → TaskState: Complete
TaskGroup의 모든 Task 완료 시 ExecuteQuestActions() 실행 (Yarn 노드 변경 등)
→ 다음 TaskGroup[1] 시작, state: Running 유지
6
배달 완료 → 퀘스트 종료
B장소 도착 후 DeliveryManager.CompleteDelivery()
→ ReceiveReport("DELIVERY_COMPLETE", B장소, 1) → TaskGroup[1] 완료
→ 마지막 TaskGroup이므로 state: WaitForCompletion → Complete, OnCompleted 이벤트 발행
핵심 — 왜 이 구조인가?
DeliveryManager는 퀘스트의 존재를 모릅니다. QuestManager는 배달의 존재를 모릅니다.
"DELIVERY_ACCEPT" 라는 문자열 카테고리만으로 두 시스템이 연결됩니다.
새 퀘스트를 추가할 때 코드를 건드리지 않고 인스펙터에서 category · target만 설정하면 됩니다. 
// QuestTask — category + target 두 조건이 모두 일치해야 반응
public bool IsTarget(string categoryStr, IdentifiedObject target)
    => HasCategory(categoryStr)
    && targets.Any(x => x.IsSameAs(target))
    && IsComplete == false;

// QuestManager — 모든 활성 퀘스트에 브로드캐스트, 매칭은 각 Task가 판단
public void ReceiveReport(string category, IdentifiedObject target, int successCount)
{
    foreach (var quest in _activeQuests.ToArray())
        quest.ReceiveReport(category, target, successCount);
}

// DeliveryManager — 퀘스트 존재를 모른 채 카테고리 문자열만 전달
public void AcceptDelivery(Location pickupLocation, DeliveryInfo info)
{
    ...
    QuestManager.Instance.ReceiveReport("DELIVERY_ACCEPT", info.PickupLocation, 1);
}
핵심 기술
기술 적용 내용
이벤트 기반 MVC Model 이벤트(OnItemPlaced/Removed)로 View가 구독 — 데이터·UI 레이어 완전 분리
2D 그리드 배치 알고리즘 아이템 크기·회전을 반영한 멀티셀 충돌 검사 및 원자적 배치/제거
하이브리드 입력 (New Input System) Mouse delta · Gamepad 버튼 감지로 실시간 모드 전환, 스틱 반복 타이머
5-State Quest FSM Inactive→Running→WaitForCompletion→Complete/Cancel, 이벤트로 UI 통보
TaskGroup 계층 구조 Quest → TaskGroup[] → QuestTask[] 3계층으로 복합 목표 순차 처리
Yarn Spinner 연동 QuestAction으로 퀘스트 진행 시 대화 분기 노드를 인스펙터에서 드롭다운 설정
